# 今天一起探讨享元模式

/**
 * 享元模式
 */
interface flyweight {
  operation(unshare: UnsharedConcreteFlyweight): string;
}

class FlyweightFactory {
  private flyweights: { [key: string]: flyweight } = {};
  getFlyweight(key: string): flyweight {
    if (!this.flyweights[key]) {
      this.flyweights[key] = new ConcreteFlyweight(key);
    }
    return this.flyweights[key];
  }
}

class ConcreteFlyweight implements flyweight {
  private intrinsicState: string;
  constructor(intrinsicState: string) {
    this.intrinsicState = intrinsicState;
  }

  operation(unshare: UnsharedConcreteFlyweight): string {
    console.log(`ConcreteFlyweight: ${this.intrinsicState}`);
    return `ConcreteFlyweight: ${this.intrinsicState} ${unshare.getName()}`;
  }
}

class UnsharedConcreteFlyweight {
  private name: string;
  constructor(name: string) {
    this.name = name;
  }
  getName(): string {
    return this.name;
  }
}

function main() {
  const factory = new FlyweightFactory();

  const fu = new UnsharedConcreteFlyweight('U');
  const fv = new UnsharedConcreteFlyweight('V');

  const fx1 = factory.getFlyweight('X');
  const res1 = fx1.operation(fu);

  console.log('res1', res1);
  const fx2 = factory.getFlyweight('X');
  const res2 = fx2.operation(fv);
  console.log('res2', res2);
}

main();
